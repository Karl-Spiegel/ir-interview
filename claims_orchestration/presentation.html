# Multi-stage claim analysis def analyze_claim(claim_text): # Stage 1: Entity Extraction entities = extract_entities(claim_text) # spaCy NER # Stage 2: Complexity Features features = { 'medical_terms': count_medical_terms(), 'legal_references': count_legal_refs(), 'damage_amount': extract_amount(), 'multiple_parties': len(entities['persons']) > 2 } # Stage 3: Classification (BERT) categories = classifier.predict(claim_text) # Stage 4: Complexity Score (0-100) complexity = calculate_complexity(features, categories) # Stage 5: Time Estimation handling_time = estimate_time(complexity, categories) return ClaimComplexity( score=complexity, categories=categories, urgency=determine_urgency(features), est_time=handling_time )
